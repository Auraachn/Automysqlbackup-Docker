Stop container

remove network = docker compose -f docker-compose.mysql-backup-test.yml down

stop aja = docker compose -f docker-compose.mysql-backup-test.yml stop

start after stop
docker compose -f docker-compose.mysql-backup-test.yml start


Pause
docker compose -f docker-compose.mysql-backup-test.yml pause

Unpause
docker compose -f docker-compose.mysql-backup-test.yml unpause

log container
docker logs -f mysql-backup

docker ps
